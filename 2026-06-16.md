script fĂĽr evenorodd.sh
````
#!/bin/bash
counter=1
while((counter <= 42))
do 
    echo $counter
    if [ $((counter % 2)) -eq 0 ]
    then
        echo "even"
    else
        echo "odd"
    fi

    ((counter++))
done

````
#!/bin/bash
sum=0
max=$1
for zahl in "$@"
do
    sum=$((sum + zahl))
    if [ $zahl -gt $max ]
    then
        max=$zahl
    fi
done
 
echo "sum: $sum"
echo "largest: $max"
