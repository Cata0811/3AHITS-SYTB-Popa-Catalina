```
time_start.sh script:

tmpfile=$(mktemp)
 
date +%s > "$tmpfile"
 
echo "$tmpfile" > .timefile
 
echo "Startzeit gespeichert."
 
time_stop.sh script:
#!/bin/bash
 
tmpfile=$(cat .timefile)
 
start=$(cat "$tmpfile")
now=$(date +%s)
 
diff=$((now - start))
 
echo "Vergangene Sekunden: $diff"
```
